Determine recursivamente si un número dado es par o impar, usando sólo la operación resta.

funcion determinar(n: entero) : logico ES
	
	
	si n = 0 entonces
		determinar:= verdadero    //par
	sino
		si n = 1 entonces
			determinar:= falso    //impar
		sino
			determinar:= determinar(n-2)
		fs
	fs
