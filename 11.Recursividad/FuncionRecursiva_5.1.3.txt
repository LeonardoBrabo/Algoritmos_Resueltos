Dados dos números: a y b. Calcule la potencia ab, usando sólo multiplicaciones sucesivas.

funcion  potencia (a,b: entero) : entero es
	si b = 1 entonces
		potencia:= a
	sino
		potencia:= a*potencia(a,b-1)
	fs
fin_funcion.
